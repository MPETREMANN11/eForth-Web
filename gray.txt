: >gray ( n -- n' )
    dup 2/ xor      \ n' = n xor ( 1 time right shift logic )
  ;

: gray> ( n -- n )
    dup 0 <> if
        0  1 31 lshift
        begin
            >r          \ save mask on return stack
            2dup 2/ xor
            r@ and or
            r> 1 rshift
            dup 0=
        until
        drop nip        \ clean data stack and leave result
    then
  ;

\ eof

: .r { value size -- }
    value <#
    size for aft # then next 
    #> type
  ;
